import {Injectable} from '@angular/core';
import {HttpClient} from "@angular/common/http";
import {Observable} from "rxjs";
import {environment} from "../../environments/environment";
import {AddCommentTopicDTO} from "../models/comments/add-comment-topic-dto";
import {GetCommentsDTO} from "../models/comments/get-comments-dto";
import {CommentEntryDTO} from "../models/comments/comment-entry-dto";
import {AddCommentDTO} from "../models/comments/add-comment-dto";

@Injectable({
  providedIn: 'root'
})
export class CommentsService {

  constructor(private httpClient: HttpClient) { }

  /*
  * This method sends a request message to get all Comments for a particular application
  */
  public getTopicsForSF328(applicationId: number): Observable<GetCommentsDTO[]> {
    const restUrl = environment.baseUrl + '/api/topics/getall/' + applicationId;
    return this.httpClient.get <GetCommentsDTO[]> (restUrl);

  }

  /*
* This method sends a request message to get all Comments for a particular question
*/
  public getTopicsFoQuestion(applicationId: number, questionId: number): Observable<GetCommentsDTO[]> {
    const restUrl = environment.baseUrl + '/api/topics/getall/' + applicationId + "/" + questionId;
    return this.httpClient.get <GetCommentsDTO[]> (restUrl);

  }

  /*
  * This method creates an RFI for a particular application
   */
  public createCommentTopic(addCommentTopicDTO:AddCommentTopicDTO): Observable<string> {
    // Construct the URL to create a Clarification request
    const restUrl = environment.baseUrl + '/api/topics/create';

    // NOTE:  The REST call is not invoked you call subscribe() on this observable
    return this.httpClient.post(restUrl,addCommentTopicDTO,{responseType: 'text'});
  }

  /*
   * This method adds a comment to a specific RFI
   */
  public addComment(newComment:AddCommentDTO): Observable<CommentEntryDTO[]> {
    const restUrl = environment.baseUrl + '/api/topics/comment';
    return this.httpClient.post<CommentEntryDTO[]> (restUrl,newComment);

  }


}
